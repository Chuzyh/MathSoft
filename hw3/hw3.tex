\begin{verbatim}
#!/bin/bash
g++ std.cpp -o std
g++ NeedCheck.cpp -o wa
for input_file in $(ls *.in); do
	./std <$input_file >out1
	./wa <$input_file >out2
	if diff out1 out2 -b -B; then
		echo "Accept" ${input_file%.*}
	else 
		echo "WrongAnswer" ${input_file%.*}
	fi
done
\end{verbatim}

    接下来运行脚本就可以看到错误代码通过了哪些测试点，在哪些测试点输出了错误的答案。
